#include <stdio.h>

void PyramidWithAsterisks() {
    int i, j, rows;
    printf("Enter rows: ");
    scanf("%d", &rows);

    for (int i = 1; i <= rows; i++){
        for (int j = 0; j < rows - i; j++){
            printf(" ");
        }
        for (int j = 1; j <= 2*i - 1; j++){
            printf("*");
        }
        printf("\n");
    }
}

int main() {
    PyramidWithAsterisks();
    return 0;
    
}
