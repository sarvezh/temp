#include <stdio.h>

void BinarySearch() {
    int key, size, middle, left, right;
    int arr[50];

    printf("Enter number to be searched:");
    scanf("%d", &key);
    printf("Enter size of array:");
    scanf("%d", &size);
    printf("Enter array elements:");
    for (int i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }
    
    left=0;
    middle=size/2;
    right=size-1;

    while (left <= right) {
        middle = (left + right)/2;
        if (arr[middle] == key) {
        printf("Element found at position %d", middle + 1);
        return;
        }
        else if (arr[middle]<key) {
            left = middle + 1;
        }
        else if (arr[middle]>key) {
            right = middle - 1;
        }       
    }
    printf("Element not found.");
}

int main() {
    BinarySearch();
    return 0;
}
