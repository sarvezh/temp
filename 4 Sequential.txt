#include <stdio.h>

void SequentialSearch() {
    int size, search;
    int arr[50];
    
    printf("Enter size of array:");
    scanf("%d", &size);
    printf("Enter number to be searched:");
    scanf("%d", &search);
    printf("Enter array elements: ");
    for (int i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    for (int i = 0; i < size; i++) {
        if (search == arr[i]) {
            printf("Number found at position %d", i + 1);
            return;
        }  
    }
    printf("Number not found.");
    return;
}

int main() {
    SequentialSearch();
    return 0;
}
