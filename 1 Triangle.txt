#include <stdio.h>

void RightAngleTriangle() {
int i, j, rows;
printf("Enter number of rows:");
scanf("%d", &rows);
    for (int i = 1; i <= rows; i++) {
        for (int j = 1; j <= i; j++) {
            printf("%d", j);
        }
        printf("\n");
    }
}

int main() {
    RightAngleTriangle();
    return 0;
}
