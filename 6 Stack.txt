#include <stdio.h>
#define SIZE 20
int stack[SIZE], top = -1, key;

int isFull(){
    if (top == SIZE - 1) {
        return 1;
    }
    else {
        return 0;
    }   
}

void Push(){
    if (top == SIZE -1) {
        printf("Stack is overflow.\n");
    }
    else {
        printf("Enter element to be pushed: ");
        scanf("%d", &key);
        top++;
        stack[top] = key; 
    }
}

int isEmpty() {
    if (top == -1) {
        return 1;
    }
    else {
        return 0;
    }
}

void Pop(){
    if (top == -1) {
        printf("Stack is underflow.\n");
    }
    else {
        printf("Popped element is: %d", stack[top]);
        top--;
    }
}

void display(){
    if (top == -1) {
        printf("Stack is underflow.\n");
    }
    else {
        printf("Stack Elements are: ");
        for (int i = 0; i <= top; i++) {
            printf("%d, ", stack[i]);
        }
    }
}

int main() {
    int n;
    do {
        printf("1. Push 2. Pop 3. Display 4. Exit");
    scanf("%d", &n);
    
    switch (n) {
        case 1: Push(); break;
        case 2: Pop(); break;
        case 3: display(); break;
        case 4: printf("Exited Menu."); break;
        default: printf("Enter valid choice.");
    }
    } while (n != 4);
    return 0;
}
